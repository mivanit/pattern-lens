24def compare_prompt_to_loaded(prompt: dict, prompt_loaded: dict) -> None:
25    for key in ("text", "hash"):
26        if prompt[key] != prompt_loaded[key]:
27            raise ActivationsMismatchError(
28                f"Prompt file does not match prompt at key {key}:\n{prompt}\n{prompt_loaded}"
29            )

32def augment_prompt_with_hash(prompt: dict) -> dict:
33    if "hash" not in prompt:
34        prompt_str: str = prompt["text"]
35        prompt_hash: str = (
36            base64.urlsafe_b64encode(hashlib.md5(prompt_str.encode()).digest())
37            .decode()
38            .rstrip("=")
39        )
40        prompt.update(hash=prompt_hash)
41    return prompt

44def load_activations(
45    model_name: str,
46    prompt: dict,
47    save_path: Path,
48) -> tuple[Path, dict[str, AttentionMatrix]]:
49    augment_prompt_with_hash(prompt)
50
51    prompt_dir: Path = save_path / model_name / "prompts" / prompt["hash"]
52    prompt_file: Path = prompt_dir / "prompt.json"
53    if not prompt_file.exists():
54        raise ActivationsMissingError(f"Prompt file {prompt_file} does not exist")
55    with open(prompt_dir / "prompt.json", "r") as f:
56        prompt_loaded: dict = json.load(f)
57        compare_prompt_to_loaded(prompt, prompt_loaded)
58
59    activations_path: Path = prompt_dir / "activations.npz"
60
61    with np.load(activations_path) as data:
62        cache = {k: torch.from_numpy(v) for k, v in data.items()}
63
64    return activations_path, cache
