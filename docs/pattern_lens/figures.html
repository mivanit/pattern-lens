 40def process_single_head(
 41    layer_idx: int,
 42    head_idx: int,
 43    attn_pattern: AttentionMatrix,
 44    save_dir: Path,
 45    force_overwrite: bool = False,
 46) -> None:
 47    for func in ATTENTION_MATRIX_FIGURE_FUNCS:
 48        fig_path: Path = save_dir / f"{func.__name__}.{FIGURE_FMT}"
 49
 50        if not force_overwrite and fig_path.exists():
 51            continue
 52
 53        try:
 54            fig, ax = plt.subplots(figsize=(10, 10))
 55            func(attn_pattern, ax)
 56            plt.tight_layout()
 57            plt.savefig(fig_path)
 58            plt.close(fig)
 59        except Exception as e:
 60            error_file = save_dir / f"{func.__name__}.error.txt"
 61            error_file.write_text(str(e))
 62            warnings.warn(
 63                f"Error in {func.__name__} for L{layer_idx}H{head_idx}: {str(e)}"
 64            )

 67def compute_and_save_figures(
 68    model_cfg: "HookedTransformerConfig|HTConfigMock",  # noqa: F821
 69    activations_path: Path,
 70    cache: dict,
 71    save_path: Path = Path(DATA_DIR),
 72    force_overwrite: bool = False,
 73) -> None:
 74    prompt_dir: Path = activations_path.parent
 75
 76    for layer_idx, head_idx in itertools.product(
 77        range(model_cfg.n_layers),
 78        range(model_cfg.n_heads),
 79    ):
 80        attn_pattern: AttentionMatrix = (
 81            cache[f"blocks.{layer_idx}.attn.hook_pattern"][0, head_idx].cpu().numpy()
 82        )
 83        save_dir: Path = prompt_dir / f"L{layer_idx}" / f"H{head_idx}"
 84        save_dir.mkdir(parents=True, exist_ok=True)
 85        process_single_head(
 86            layer_idx=layer_idx,
 87            head_idx=head_idx,
 88            attn_pattern=attn_pattern,
 89            save_dir=save_dir,
 90            force_overwrite=force_overwrite,
 91        )
 92
 93    generate_prompts_jsonl(save_path / model_cfg.model_name)

117def figures_main(
118    model_name: str,
119    save_path: str,
120    n_samples: int,
121    force: bool,
122    parallel: bool | int = True,
123) -> None:
124    with SpinnerContext(message="setting up paths", **SPINNER_KWARGS):
125        # save model info or check if it exists
126        save_path: Path = Path(save_path)
127        model_path: Path = save_path / model_name
128        with open(model_path / "model_cfg.json", "r") as f:
129            model_cfg = HTConfigMock.load(json.load(f))
130
131    with SpinnerContext(message="loading prompts", **SPINNER_KWARGS):
132        # load prompts
133        with open(model_path / "prompts.jsonl", "r") as f:
134            prompts: list[dict] = [json.loads(line) for line in f.readlines()]
135        # truncate to n_samples
136        prompts = prompts[:n_samples]
137
138    print(f"{len(prompts)} prompts loaded")
139
140    list(
141        run_maybe_parallel(
142            func=functools.partial(
143                process_prompt,
144                model_cfg=model_cfg,
145                save_path=save_path,
146                force_overwrite=force,
147            ),
148            iterable=prompts,
149            parallel=parallel,
150            pbar="tqdm",
151            pbar_kwargs=dict(
152                desc="Making figures",
153                ascii=" =",
154            ),
155        )
156    )
157
158    with SpinnerContext(
159        message="updating jsonl metadata for models and functions", **SPINNER_KWARGS
160    ):
161        generate_models_jsonl(save_path)
162        generate_functions_jsonl(save_path)

165def main():
166    with SpinnerContext(message="parsing args", **SPINNER_KWARGS):
167        arg_parser: argparse.ArgumentParser = argparse.ArgumentParser()
168        # input and output
169        arg_parser.add_argument(
170            "--model",
171            "-m",
172            type=str,
173            required=True,
174            help="The model name to use",
175        )
176        arg_parser.add_argument(
177            "--save-path",
178            "-s",
179            type=str,
180            required=False,
181            help="The path to save the attention patterns",
182            default=DATA_DIR,
183        )
184        # number of samples
185        arg_parser.add_argument(
186            "--n-samples",
187            "-n",
188            type=int,
189            required=False,
190            help="The max number of samples to process, do all in the file if None",
191            default=None,
192        )
193        # force overwrite of existing figures
194        arg_parser.add_argument(
195            "--force",
196            "-f",
197            type=bool,
198            required=False,
199            help="Force overwrite of existing figures",
200            default=False,
201        )
202
203        args: argparse.Namespace = arg_parser.parse_args()
204
205    print(f"args parsed: {args}")
206
207    figures_main(
208        model_name=args.model,
209        save_path=args.save_path,
210        n_samples=args.n_samples,
211        force=args.force,
212    )
